#include <stdio.h>
#include <stdlib.h>

typedef struct a{
	unsigned char intensity;
	unsigned int freq;
	double prob;
	struct a *next;	
}node;

void main(){
	int i=0, j=0, rowSize = 140, colSize = 320, n=0, maxIntens=10, maxfreq=0, cnt =0;;
	char fname [] = "csm.raw";
	FILE *f;
	unsigned char *intensityArray, data, *array10;
	node *head, *tmp, *cur;


	if( (f=fopen(fname,"rb")) == NULL ) {printf("error\n"); exit (1);}

	intensityArray = (unsigned char*)calloc(sizeof(unsigned char),255);

	for(i=0; i<rowSize; i++){
		for(j=0; j<colSize; j++){
			fread(&data,1,1,f);
			intensityArray[data]++;
		}
	}

	for(i=0; i<255; i++){
		if(intensityArray[i]>10) n++;//10º¸´Ù Å« °ªÀÇ °³¼ö -> n
	}


	array10 = (unsigned char*)malloc(sizeof(unsigned char)*n);

	j=0;
	for(i=0; i<255; i++){
		if(intensityArray[i]>10) {
			array10[j++] = i; //array10¿¡´Â ¹à±â°ªÀÌ µé¾î°¡ÀÖÀ½
		}
	}

	
	
	maxfreq = intensityArray[ array10[0] ];
	maxIntens = array10[0];
	for(i=0; i<n; i++){
		if(maxfreq < intensityArray[ array10[i] ]) {
			maxfreq = intensityArray[ array10[i] ];
			maxIntens = array10[i];
		}
	}

	cur = (node*)malloc(sizeof(node));
	cur->intensity = maxIntens;
	cur->freq = intensityArray[maxIntens];
	cur->prob = cur->freq/(double)rowSize/colSize;
	cur->next = NULL;
	head = tmp = cur;

	printf("¹à±â°ª : %d\nºóµµ¼ö : %d\nÈ®·ü : %.2f\n------------\n",cur->intensity,cur->freq,cur->prob);


	i=0;
	while(i != n-1){ 
		if( array10[i] == maxIntens ) i++;
		cur = (node*)malloc(sizeof(node));
		cur->intensity = array10[i];
		cur->freq = intensityArray[ array10[i] ];
		cur->prob = cur->freq/(double)rowSize/colSize;
		cur->next = NULL;

		if(tmp->next == NULL) {
			tmp->next = cur; 
			tmp = head;
			i++;
		}
		else if(tmp->freq > cur->freq && tmp->next->freq > cur->freq) {
			tmp = tmp->next; 
			free(cur);

		}
		else if(tmp->freq > cur->freq && tmp->next->freq <= cur->freq) {
			cur->next = tmp->next;
			tmp->next = cur;
			tmp = head;
			i++;	
		}
	}

	for(i=0; i<10; i++){
		printf("¹à±â°ª : %d\nºóµµ¼ö : %d\nÈ®·ü : %.2f\n------------\n",tmp->intensity,tmp->freq,tmp->prob);
		tmp= tmp->next;
	}

	printf("-------------------------------------------------------------\n");

	tmp = head;
	while(cnt != 10){
		if(tmp->next->intensity >= 100 && tmp->next->intensity<=120){
			cnt++;
			cur = tmp->next;
			tmp->next = tmp->next->next;
			free(cur);
		}
		else tmp = tmp->next;
	}

	tmp = head;
	for(i=0; i<10; i++){
		printf("¹à±â°ª : %d\nºóµµ¼ö : %d\nÈ®·ü : %.2f\n------------\n",tmp->intensity,tmp->freq,tmp->prob);
		tmp= tmp->next;
	}

	tmp = head;
	while(tmp != NULL){
		tmp = head->next;
		free(head);
		head = tmp;
	}


}
